​    从理论上讲：快排速度会高于堆排

1. 快排移动指针都是连续的，但是堆排每次都是父子节点进行比较，它们不是连续的，不能很好的利用CPU的缓存机制
2. 堆排的堆化操作会打乱本来可能已经有序的序列，所以比较操作比快排多

# 快速排序

```java
public class QuickSort {
    public static void sort(int[] arr){
        if(arr==null||arr.length<=1)
            return;
        sort(arr,0,arr.length-1);
    }

    private static void sort(int[] arr, int left, int right) {
        if(left>=right)
            return;
        int pivotPos=partition(arr,left,right);
        sort(arr,left,pivotPos-1);
        sort(arr,pivotPos+1,right);
    }

    private static int partition(int[] arr, int left, int right) {
        int pivot=arr[left];
        while (left<right){
            //必须从右边开始
            while (arr[right]>=pivot&&left<right) right--;
            arr[left]=arr[right];
            while (arr[left]<=pivot&&left<right)left++;
            arr[right]=arr[left];
        }
        arr[left]=pivot;
        return left;
    }

```

# 堆排序

```java
public class HeapSort {
    public static void sort(int[] arr){
        int len=arr.length-1;
        buildMaxHeap(arr,len);
        for(int i=0;i<len-1;i++){
            int temp=arr[1];
            arr[1]=arr[len-i];
            arr[len-i]=temp;
            adjustDown(arr,1,len-i-1);
        }
    }
    //建堆
    private static void buildMaxHeap(int[] arr,int len) {
        for(int i=len/2;i>=1;i--){
            adjustDown(arr,i,len);
        }
    }
//调整
    private static void adjustDown(int[] arr, int parent,int len) {
        arr[0]=arr[parent];
        for(int child=parent*2;child<=len;child*=2){
            if(child+1<=len&&arr[child]<arr[child+1]){
                child++;
            }
            if(arr[0]>=arr[child]){//注意这里一定是arr[0]，之前写成了arr[parent]调了好久。。。
                break;
            }
            else {
                arr[parent]=arr[child];
                parent=child;
            }
        }
        arr[parent]=arr[0];
    }
}
```



# 归并排序

```java
public class MergeSort {
    public static void sort(int[] arr){
        sort(arr,0,arr.length-1);
    }

    private static void sort(int[] arr, int low, int high) {
        if(low<high){
            int mid=(low+high)>>1;
            sort(arr,low,mid);//左边排序
            sort(arr,mid+1,high);//右边排序
            merge(arr,low,mid,high);
        }
    }

    private static void merge(int[] arr, int low, int mid, int high) {
        int[] temp=new int[high-low+1];
        int k=0;
        int i=low,j=mid+1;
        while (i<=mid&&j<=high){
            if(arr[i]<=arr[j]){
                temp[k++]=arr[i++];
            }else {
                temp[k++]=arr[j++];
            }
        }
        while (i<=mid){
            temp[k++]=arr[i++];
        }
        while (j<=high){
            temp[k++]=arr[j++];
        }
        for(i=0;i<k;i++){
            arr[low+i]=temp[i];
        }
    }
}
```

public class ShellSort {
   public static void sort(int[] arr){
       if(arr==null||arr.length<=1)return;
        for(int gap=arr.length/2;gap!=0;gap/=2){
            for(int i=gap;i<arr.length;i++){
                int j;
                int current=arr[i];
                for(j=i-gap;j>=0;j-=gap){
                    if(current>arr[j])
                        break;
                    arr[j+gap]=arr[j];
                }
                arr[j+gap]=current;
            }
        }
   }

}

# 插入排序

``` java
public class InsertSort {
    public static void sort(int[] arr) {
        if (arr == null || arr.length <= 1)
            return;
        //要插入的次数
        for (int i = 0; i < arr.length - 1; i++) {
            int current = arr[i + 1];
            int j;
            for (j = i; j >= 0; j--) {
                if (current >= arr[j]) {
                    break;
                }
                arr[j + 1] = arr[j];
            }
            arr[j + 1] = current;
        }
    }
}
```

# 冒泡排序

```java
public class BubbleSort {
    public static void sort(int[] arr){
        if(arr==null||arr.length<=1) return;
         for(int i=0;i<arr.length-1;i++){
             boolean flg=true;
             for(int j=0;j<arr.length-1-i;j++){
                if(arr[j]>arr[j+1]){
                    int temp=arr[j];
                    arr[j]=arr[j+1];
                    arr[j+1]=temp;
                    flg=false;
                }
             }
             if(flg==true)
                 break;
         }
    }
}
```

# 选择排序

```java
public class SelectSort {
    public static void sort(int[] arr) {
        if (arr == null || arr.length <= 1)
            return;
        for (int i = 0; i < arr.length - 1; i++) {
            int min = arr[i];
            int index = i;
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < min) {
                    min=arr[j];
                    index=j;
                }
            }
            int temp=arr[i];
            arr[i]=min;
            arr[index]=temp;
        }
    }
}
```
